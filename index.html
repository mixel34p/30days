📦 SQL — CREAR TABLAS EN SUPABASE

Ejecuta el siguiente SQL en el editor de Supabase (SQL Editor): 

-- ═══════════════════════════════════════════════
-- Tabla: users
-- ═══════════════════════════════════════════════
CREATE TABLE IF NOT EXISTS public.users (
  id          UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  nickname    TEXT NOT NULL UNIQUE,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Index for fast lookup by nickname
CREATE INDEX IF NOT EXISTS idx_users_nickname ON public.users(nickname);

-- Row Level Security: anyone can read users, insert their own
ALTER TABLE public.users ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Anyone can read users"
  ON public.users FOR SELECT USING (true);

CREATE POLICY "Anyone can insert user"
  ON public.users FOR INSERT WITH CHECK (true);

-- ═══════════════════════════════════════════════
-- Tabla: responses
-- ═══════════════════════════════════════════════
CREATE TABLE IF NOT EXISTS public.responses (
  id             UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_nickname  TEXT NOT NULL REFERENCES public.users(nickname) ON DELETE CASCADE,
  day_number     INTEGER NOT NULL CHECK (day_number BETWEEN 1 AND 30),
  game_id        INTEGER,
  game_name      TEXT,
  game_cover     TEXT,
  character_name TEXT,
  text_response  TEXT,
  created_at     TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE (user_nickname, day_number)
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_responses_day    ON public.responses(day_number);
CREATE INDEX IF NOT EXISTS idx_responses_user   ON public.responses(user_nickname);

-- RLS: anyone can read, users can upsert their own
ALTER TABLE public.responses ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Anyone can read responses"
  ON public.responses FOR SELECT USING (true);

CREATE POLICY "Anyone can insert response"
  ON public.responses FOR INSERT WITH CHECK (true);

CREATE POLICY "Anyone can update response"
  ON public.responses FOR UPDATE USING (true);

📋 PASOS PARA DESPLEGAR EN VERCEL
  1. Crea una cuenta en vercel.com
  2. Importa este repositorio desde GitHub
  3. En Settings → Environment Variables, añade las 4 variables de entorno
  4. En el archivo index.html, rellena también window.ENV_SUPABASE_URL y window.ENV_SUPABASE_ANON_KEY (son claves públicas, es seguro)
  5. Despliega → ¡listo!
